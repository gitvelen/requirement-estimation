### 测试用例1：简单需求评估

#### 1.1 测试文档

**文档名称**: `简单需求-反洗钱系统客户评级.docx`

**需求内容**：
```
需求名称：反洗钱系统微粒贷客户风险评级需求

需求简述：微粒贷客户应纳入客户风险评级中

需求内容说明：
由于微粒贷客户信息不能完整，无法满足在我行ECIF中开立客户编号所要求的九要素。
为减低对现有系统的改动以及客户信息补全的不确定性，微粒贷这批客户不在ECIF中建立客户编号，
统一由数据仓库卸数给反洗钱系统后，反洗钱系统通过为微粒贷客户单独建立的客户风险评级模型进行客户风险等级评级。
```

#### 1.2 预期结果

**识别的系统**：
- 反洗钱系统
- 数据仓库

**功能点拆分**（反洗钱系统）：
- 微粒贷客户数据获取（2人天）
- 客户数据清洗转换（3人天）
- 评级规则配置（5人天）
- 评级计算引擎（8人天）
- 评级结果存储（2人天）
- 评级报告生成（3人天）

**总工作量**：约23人天

#### 1.3 测试步骤

```bash

# 4. 等待完成（约30-60秒）

# 5. 下载报告

  -o "简单需求评估报告.xlsx"
```

#### 1.4 验证点

- ✅ 系统识别准确（至少识别出反洗钱系统）
- ✅ 功能点拆分合理（粒度0.5-5人天）

## 七、自动化测试脚本

### 完整测试脚本

```python
import requests
import time
import os

class TestClient:
    """测试客户端"""

    def __init__(self, base_url="http://localhost:8000"):
        self.base_url = base_url

    def test_health_check(self):
        """测试健康检查"""
        print("测试1: 健康检查")
        response = requests.get(f"{self.base_url}/api/v1/health")
        assert response.status_code == 200
        print("✓ 健康检查通过")

    def test_upload_simple_requirement(self):
        """测试简单需求上传"""
        print("\n测试2: 上传简单需求")
        file_path = "tests/fixtures/简单需求-反洗钱系统客户评级.docx"

        if not os.path.exists(file_path):
            print(f"✗ 测试文件不存在: {file_path}")
            return

        files = {"file": open(file_path, "rb")}
        response = requests.post(f"{self.base_url}/api/v1/requirement/upload", files=files)

        assert response.status_code == 200
        result = response.json()
        assert result["code"] == 200

        task_id = result["data"]["task_id"]
        print(f"✓ 上传成功，任务ID: {task_id}")

        return task_id

    def test_wait_completion(self, task_id, timeout=300):
        """等待任务完成"""
        print(f"\n测试3: 等待任务完成")

        start_time = time.time()
        while time.time() - start_time < timeout:
            response = requests.get(f"{self.base_url}/api/v1/requirement/status/{task_id}")
            result = response.json()
            status = result["data"]["status"]
            progress = result["data"]["progress"]

            print(f"  状态: {status}, 进度: {progress}%")

            if status == "completed":
                print("✓ 任务完成")
                return True
            elif status == "failed":
                print(f"✗ 任务失败: {result['data'].get('error')}")
                return False

            time.sleep(5)

        print("✗ 任务超时")
        return False

    def test_download_report(self, task_id):
        """测试下载报告"""
        print(f"\n测试4: 下载报告")

        response = requests.get(f"{self.base_url}/api/v1/requirement/report/{task_id}")

        assert response.status_code == 200
        assert response.headers["content-type"] == "application/vnd.openxmlformats-officedocument.spreadsheetml.sheet"

        save_path = f"tests/output/{task_id}_report.xlsx"
        os.makedirs(os.path.dirname(save_path), exist_ok=True)

        with open(save_path, "wb") as f:
            f.write(response.content)

        print(f"✓ 报告已下载: {save_path}")
        return True

    def run_all_tests(self):
        """运行所有测试"""
        print("=" * 60)
        print("开始自动化测试")
        print("=" * 60)

        try:
            self.test_health_check()
            task_id = self.test_upload_simple_requirement()

            if task_id:
                if self.test_wait_completion(task_id):
                    self.test_download_report(task_id)

            print("\n" + "=" * 60)
            print("所有测试通过 ✓")
            print("=" * 60)

        except AssertionError as e:
            print(f"\n✗ 测试失败: {str(e)}")
        except Exception as e:
            print(f"\n✗ 测试异常: {str(e)}")

if __name__ == "__main__":
    client = TestClient()
    client.run_all_tests()
```
